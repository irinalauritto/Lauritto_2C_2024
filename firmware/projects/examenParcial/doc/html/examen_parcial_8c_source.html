259 //Ademas se divide por mil ya que estamos midiendo en mV y necesitamos las medidas en V.
260
261 medicionX = (float)medicionX/1000;
262 medicionY = (float)medicionY/1000;
263 medicionZ =(float)medicionZ/1000;
264

266 float GenY = (medicionY-1.65)/0.3;
267 float GenZ = (medicionZ-1.65)/0.3;
268
269 float GTotal = GenX+GenY+GenZ;
